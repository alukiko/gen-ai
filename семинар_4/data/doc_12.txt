Вариант 1. Самый простой: app.stream(...) и печать изменений state

Этот способ показывает, что меняется после каждого узла (planner/oracle/search/answer и т.д.).

from pprint import pprint

def run_with_trace(app, initial_state):
 last_resources = None

for event in app.stream(initial_state, stream_mode="values"):
 # event — это текущее значение state после шага
 resources = event.get("resources")
 if resources and resources != last_resources:
 print("\n=== RESOURCES UPDATED ===")
 pprint(resources)
 last_resources = resources

# если ты хранишь выбор модели внутри intermediate_results — тоже можно печатать
 # pprint(event.get("intermediate_results"))

return event # последнее состояние

final_state = run_with_trace(app, {
 "query": "What is the official website of the University of Oxford?",
 "tasks": [],
 "current_task_id": None,
 "resources": {},
 "intermediate_results": {},
 "final_answer": None,
})
print("\nFINAL ANSWER:", final_state.get("final_answer"))

✅ плюс: без изменения графа
✅ увидишь, когда oracle записал ресурсы
⚠️ минус: если узлы перезаписывают resources, увидишь только последнее значение

Вариант 3. Самый надёжный: логировать выбор модели прямо в state (рекомендую)

Это лучший способ для твоего исследования: ты сам сохраняешь “allocation decisions” Oracle.

В oracle-узле добавь:

def oracle_allocate_resources(state: GraphState) -> GraphState:
 # ... твоя логика выбора ресурсов
 state["resources"] = resources

# лог решения оракула
 trace = state.get("allocation_trace", [])
 trace.append({
 "step": len(trace) + 1,
 "resources": resources,
 })

return state

И потом печатай:

✅ плюс: ты получаешь историю решений, а не только финальный resources
✅ идеально для анализа fairness

from __future__ import annotations
from typing import Any, Dict, List, Optional, TypedDict
import re
from urllib.parse import urlparse

# Если у тебя уже есть тип GraphState — используй его.
class GraphState(TypedDict, total=False):
 query: str
 tasks: List[Dict[str, Any]]
 current_task_id: Optional[str]
 resources: Dict[str, Dict[str, Any]]
 intermediate_results: Dict[str, Any]
 final_answer: Optional[str]

def _safe_get(d: Dict[str, Any], key: str, default=None):
 v = d.get(key, default)
 return default if v is None else v

def _extract_urls(text: str) -> List[str]:
 if not text:
 return []
 # простая эвристика для URL
 urls = re.findall(r"(https?://[^\s\]\)\"'>]+)", text)
 # очистка хвостов
 cleaned = []
 for u in urls:
 u = u.rstrip(".,;:)")
 try:
 p = urlparse(u)
 if p.scheme in ("http", "https") and p.netloc:
 cleaned.append(u)
 except Exception:
 pass
 # unique preserving order
 seen = set()
 out = []
 for u in cleaned:
 if u not in seen:
 seen.add(u)
 out.append(u)
 return out

def _merge_intermediate(state: GraphState, task_id: str, payload: Any) -> None:
 if "intermediate_results" not in state or state["intermediate_results"] is None:
 state["intermediate_results"] = {}
 state["intermediate_results"][task_id] = payload
1) Extraction / Parsing Agent (роль extract)

Задача: взять сырой текст/ссылки из intermediate_results (обычно от Search Agent) и вернуть структурированные факты: ссылки, числа, даты, кандидаты-ответы.

def extraction_agent(task: Dict[str, Any], state: GraphState, resource_cfg: Dict[str, Any]) -> Dict[str, Any]:
 """
 Минимальный extractor:
 - ищет источники в предыдущих результатах (facts/source_url/snippet/raw_text)
 - извлекает URL и потенциальные "ключевые значения" (числа/даты/краткие тезисы)
 """
 task_id = task["id"]
 desc = _safe_get(task, "description", "")

# Попробуем найти "входные данные" для извлечения:
 # 1) явный список в task["input_task_ids"] (если ты так сделаешь)
 input_task_ids = task.get("input_task_ids") or task.get("depends_on") or []
 blobs: List[str] = []

for dep_id in input_task_ids:
 dep_res = (state.get("intermediate_results") or {}).get(dep_id)
 if not dep_res:
 continue
 # ожидаемые поля от search_agent: summary / facts[*].fact / facts[*].source_url
 if isinstance(dep_res, dict):
 if "summary" in dep_res and dep_res["summary"]:
 blobs.append(str(dep_res["summary"]))
 if "facts" in dep_res and isinstance(dep_res["facts"], list):
 for f in dep_res["facts"]:
 if isinstance(f, dict):
 if f.get("fact"):
 blobs.append(str(f["fact"]))
 if f.get("source_url"):
 blobs.append(str(f["source_url"]))
 if "raw_text" in dep_res and dep_res["raw_text"]:
 blobs.append(str(dep_res["raw_text"]))
 else:
 blobs.append(str(dep_res))

# Также можно извлечь из общего query/описания
 blobs.append(state.get("query", ""))
 blobs.append(desc)

full_text = "\n".join([b for b in blobs if b])

urls = _extract_urls(full_text)

# числа/даты — грубые эвристики, пригодные для GAIA L1/L2 и дебага
 numbers = re.findall(r"\b\d+(?:[.,]\d+)?\b", full_text)
 years = re.findall(r"\b(19\d{2}|20\d{2})\b", full_text)

# короткие "кандидаты фактов": предложения с числами
 candidates = []
 for sent in re.split(r"(?<=[.!?])\s+", full_text):
 if re.search(r"\b\d+(?:[.,]\d+)?\b", sent):
 s = sent.strip()
 if 10 <= len(s) <= 240:
 candidates.append(s)
 if len(candidates) >= 10:
 break

result = {
 "status": "done",
 "task_id": task_id,
 "input_task_ids": input_task_ids,
 "extracted": {
 "urls": urls,
 "numbers": numbers[:30],
 "years": years[:30],
 "candidate_sentences": candidates,
 },
 "summary": f"Extracted {len(urls)} urls, {len(numbers)} numbers, {len(years)} years.",
 }

_merge_intermediate(state, task_id, result)
 return result
2) Reasoning Agent (роль reason)

Задача: взять структурированные факты от extractor/search и сформировать вывод/под-ответ, возможно — гипотезу финального ответа.

def reasoning_agent(task: Dict[str, Any], state: GraphState, resource_cfg: Dict[str, Any]) -> Dict[str, Any]:
 """
 Минимальный reasoner (без внешнего LLM):
 - собирает факты из depends_on
 - делает простую агрегацию и предлагает "draft_answer"
 Если хочешь — потом заменишь внутреннюю логику на вызов LLM.
 """
 task_id = task["id"]
 desc = _safe_get(task, "description", "")

input_task_ids = task.get("input_task_ids") or task.get("depends_on") or []
 deps = [(state.get("intermediate_results") or {}).get(dep_id) for dep_id in input_task_ids]
 deps = [d for d in deps if d]

# Соберём URL-кандидаты и факты
 urls: List[str] = []
 fact_lines: List[str] = []

for d in deps:
 if isinstance(d, dict):
 # extractor output
 extracted = d.get("extracted")
 if isinstance(extracted, dict):
 urls.extend(extracted.get("urls") or [])
 for s in (extracted.get("candidate_sentences") or []):
 fact_lines.append(str(s))
 # search output
 if "facts" in d and isinstance(d["facts"], list):
 for f in d["facts"]:
 if isinstance(f, dict) and f.get("fact"):
 fact_lines.append(str(f["fact"]))
 if isinstance(f, dict) and f.get("source_url"):
 urls.append(str(f["source_url"]))
 if d.get("summary"):
 fact_lines.append(str(d["summary"]))
 else:
 fact_lines.append(str(d))

# простая эвристика: если вопрос про "official website" — берём первый нормальный url
 q = (state.get("query") or "") + " " + desc
 draft_answer = None
 if re.search(r"\bofficial\b|\bwebsite\b|\bsite\b|\bсайт\b|\bофициальн", q, re.IGNORECASE):
 if urls:
 draft_answer = urls[0]

# fallback: просто кратко агрегируем
 if draft_answer is None:
 draft_answer = (fact_lines[0] if fact_lines else "").strip()[:300] or "No draft answer."

result = {
 "status": "done",
 "task_id": task_id,
 "input_task_ids": input_task_ids,
 "reasoning": {
 "notes": fact_lines[:12],
 "candidate_urls": urls[:10],
 "draft_answer": draft_answer,
 },
 "summary": "Produced a draft answer from intermediate evidence.",
 }

_merge_intermediate(state, task_id, result)
 return result

3) Calculation Agent (роль calc)

Задача: выполнить расчёт/код. Для безопасности лучше разрешать только простые выражения или “мини-питон” в изоляции.

Ниже — безопасный калькулятор для базовой математики (GAIA часто требует именно такого).

import ast
import operator as op

_ALLOWED_OPS = {
 ast.Add: op.add,
 ast.Sub: op.sub,
 ast.Mult: op.mul,
 ast.Div: op.truediv,
 ast.Pow: op.pow,
 ast.USub: op.neg,
 ast.Mod: op.mod,
}

def _safe_eval_expr(expr: str) -> float:
 """
 Безопасная оценка арифметики: + - * / ** % и скобки.
 Никаких имён, вызовов, атрибутов, импортов.
 """
 def _eval(node):
 if isinstance(node, ast.Num): # py<3.8
 return node.n
 if isinstance(node, ast.Constant) and isinstance(node.value, (int, float)):
 return node.value
 if isinstance(node, ast.BinOp) and type(node.op) in _ALLOWED_OPS:
 return _ALLOWED_OPS[type(node.op)](_eval(node.left), _eval(node.right))
 if isinstance(node, ast.UnaryOp) and type(node.op) in _ALLOWED_OPS:
 return _ALLOWED_OPS[type(node.op)](_eval(node.operand))
 raise ValueError("Unsupported expression")

tree = ast.parse(expr, mode="eval")
 return float(_eval(tree.body))

def calculation_agent(task: Dict[str, Any], state: GraphState, resource_cfg: Dict[str, Any]) -> Dict[str, Any]:
 """
 Минимальный calc-agent:
 - ищет выражение в task["expression"] либо пытается вытащить из description/query
 - считает безопасно
 """
 task_id = task["id"]
 desc = _safe_get(task, "description", "")

expr = task.get("expression")
 if not expr:
 # попытка вытащить из описания/запроса
 text = f"{state.get('query','')} {desc}"
 # очень простая эвристика: берем то, что похоже на мат. выражение
 m = re.search(r"([-+*/()%\d\s\.]+)", text)
 expr = m.group(1).strip() if m else ""

expr = (expr or "").strip()
 if not expr:
 result = {
 "status": "failed",
 "task_id": task_id,
 "error": "No expression to calculate.",
 }
 _merge_intermediate(state, task_id, result)
 return result

try:
 value = _safe_eval_expr(expr)
 result = {
 "status": "done",
 "task_id": task_id,
 "expression": expr,
 "value": value,
 "summary": f"Calculated expression = {value}",
 }
 except Exception as e:
 result = {
 "status": "failed",
 "task_id": task_id,
 "expression": expr,
 "error": str(e),
 }

_merge_intermediate(state, task_id, result)
 return result

def route_by_role(state: GraphState) -> str:
 current_id = state.get("current_task_id")
 if current_id is None:
 return "finish"

task = next(t for t in state["tasks"] if t["id"] == current_id)
 role = task["role"]

if role == "search":
 return "search_agent"
 elif role in ("extract", "parse", "extraction"):
 return "extraction_agent"
 elif role in ("reason", "reasoning"):
 return "reasoning_agent"
 elif role in ("calc", "calculation", "code"):
 return "calculation_agent"
 elif role in ("critic", "verify", "verification"):
 return "critic_agent"
 elif role in ("answer", "summarize", "report"):
 return "answer_agent"
 else:
 return "finish"

“Using the attached PDF, determine …”

➡️ Ассистент должен открыть файл, извлечь информацию и использовать её в ответе.

3️⃣ Задачи, требующие вычислений

Здесь LLM без кода часто ошибается.

Примеры:

подсчёты,
агрегаты,
статистика,
работа с таблицами,
числовые сравнения.

➡️ Именно под это в GAIA нужен Calculation Agent / PythonREPLTool.

4️⃣ Многошаговые (multi-hop) задачи

Задачи, где:

нужно что-то найти,
потом извлечь нужные данные,
затем рассудить,
иногда — посчитать,
и только потом ответить.

➡️ Это основной сценарий GAIA.

5️⃣ Инструментальные задачи (tool-use)

GAIA специально проверяет:

умеет ли система решать, КОГДА нужен инструмент,
умеет ли не использовать инструмент, когда он не нужен.

Примеры:

поиск в интернете,
работа с файлами,
вычисления,
проверка фактов.

➡️ Это ровно твоя архитектура:
Planner → Oracle → Agents.

🔹 Как это отражается в датасете

Типичные поля в примере GAIA:

Question — текст вопроса
file_path — путь к вложению (может быть None)
Level — уровень сложности (1 / 2 / 3)
Final answer — эталонный ответ (только в validation)

Если file_path ≠ None, значит:

❗ Задача НЕ чисто текстовая

🔹 Почему твоя система подходит для GAIA
Возможность GAIA Агент в твоей системе
Поиск информации search
Извлечение из файлов extract
Логический вывод reason
Вычисления calc
Проверка critic
Финальный ответ answer

GAIA фактически спроектирован под мультиагентные пайплайны, а не под “один LLM”.

В твоём текущем валидаторе мы делаем так:

 query = f"{question}\n\n[Attachment file available at path: {abs_file_path}]"

что тогда делает эта агент тула, если не умеет читать документы?

import json
from langchain_core.messages import SystemMessage, HumanMessage

def extraction_agent_node(state: GraphState) -> GraphState:
 task_id = state.get("current_task_id")
 if not task_id:
 return state

task = next(t for t in state["tasks"] if t["id"] == task_id)
 cfg = state["resources"].get(task_id, {})
 state.setdefault("intermediate_results", {})

# Берём всё, что накопили агенты (обычно после Search)
 intermediate = state["intermediate_results"]

llm = make_llm(
 model=cfg.get("model", "openai/gpt-4o-mini"),
 temperature=cfg.get("temperature", 0.0),
 max_tokens=cfg.get("max_tokens", 900),
 tools=[],
 )

system_prompt = (
 "You are an Extraction/Parsing agent.\n"
 "Extract structured information from the provided intermediate results.\n"
 "Output STRICT JSON only.\n"
 )

user_payload = {
 "task_description": task["description"],
 "intermediate_results": intermediate,
 "required_output_schema": {
 "entities": [{"name": "str", "type": "str", "evidence": "str"}],
 "key_points": ["str"],
 "numbers_and_units": [{"value": "str", "unit": "str", "context": "str"}],
 "definitions": [{"term": "str", "definition": "str", "source_url": "optional str"}],
 "open_questions": ["str"]
 }
 }

resp = llm.invoke([
 SystemMessage(content=system_prompt),
 HumanMessage(content=json.dumps(user_payload, ensure_ascii=False))
 ])

try:
 parsed = json.loads(resp.content)
 except Exception:
 parsed = {"_raw_text": resp.content}

task["result"] = parsed
 task["status"] = "done"
 state["intermediate_results"][task_id] = parsed
 return state

Отличный и очень правильный вопрос.
Коротко: этот агент — не “читатель документов”, а “нормализатор / структуризатор информации”. Он никогда не задумывался как File Reader — и это нормально.

Ниже — чёткое объяснение что он реально делает, зачем он нужен и где его граница ответственности.

1️⃣ Что этот Extraction Agent ДЕЛАЕТ на самом деле

Этот агент работает НЕ с сырыми документами, а с уже полученной информацией.

Он:

берёт неструктурированные результаты других агентов;
превращает их в машиночитаемую структуру (JSON);
выделяет:
сущности,
определения,
числа,
ключевые пункты,
открытые вопросы.
Его вход:
intermediate_results = {
 "T1": { "facts": [...], "sources": [...] }, # search
 "T2": "...", # reason
}
Его выход:
{
 "entities": [...],
 "definitions": [...],
 "numbers_and_units": [...],
 "key_points": [...],
 "open_questions": [...]
}

➡️ Он НЕ читает PDF, НЕ парсит CSV, НЕ открывает файлы.

2️⃣ Зачем он нужен, если не читает документы

Потому что в мультиагентных системах есть разница между:

Шаг Что происходит
Search собираем шумные факты
Extract делаем структуру
Reason думаем
Calc считаем
Answer пишем
Без extraction:
reasoning агент получает кашу;
answer агент склонен галлюцинировать;
critic сложнее проверять логику.

Extraction — это контракт между агентами.

3️⃣ Почему он не читает файлы — и это правильно

Чтение файлов — другая ответственность:

Агент Ответственность
FileReader открыть PDF / CSV / TXT
Extract структурировать текст
Calc вычислить
Reason вывести
Answer оформить

Если объединить чтение + extraction:

агент становится слишком тяжёлым;
сложнее контролировать инструменты;
хуже масштабируется.

GAIA как раз разделяет эти навыки.

4️⃣ Что происходит в твоей системе СЕЙЧАС

Сейчас цепочка выглядит так:

search → extract → reason → calc → critic → answer

Extraction:

работает корректно,
но только для text-based источников (search results, summaries).

Поэтому:

text-only GAIA задачи → ✅
file-based GAIA задачи → ⚠️ (нужно добавить FileReader)
5️⃣ Что нужно добавить, чтобы Extraction работал с документами
Минимальное расширение (рекомендуется):
➕ FileReader Agent

Агент, который:

принимает file_path,
читает файл,
кладёт сырой текст / таблицу в intermediate_results.
search → fileread → extract → ...

Пример (идея):

state["intermediate_results"]["FILE_TEXT"] = extracted_text

После этого твой Extraction Agent начнёт работать “на полную”, потому что у него появится текст для парсинга.

6️⃣ Очень важный концептуальный вывод (для отчёта)

Ты можешь честно написать:

The Extraction agent is designed to normalize and structure intermediate textual outputs produced by other agents, rather than directly processing raw documents. Document parsing is treated as a separate tool-oriented capability.

Это правильная инженерная декомпозиция, а не недостаток.

7️⃣ Ответ одной строкой

Extraction Agent не читает документы, он превращает уже полученный текст в структуру. Для чтения документов нужен отдельный FileReader Agent.
